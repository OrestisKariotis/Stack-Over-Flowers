@startuml

database "Elastic Search" {
    [events]
}
interface eventsInterface
events -- eventsInterface


component Backend {
       [providersReport]
}
interface providersReportInterface
providersReport -- providersReportInterface
providersReport --> eventsInterface


component Frontend {
      [monthlyReportScript]
}
interface monthlyReportInterface
monthlyReportScript -- monthlyReportInterface
monthlyReportScript --> providersReportInterface

:Provider: as p
p -u-> monthlyReportInterface

      
@enduml
